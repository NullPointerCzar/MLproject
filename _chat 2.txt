% Time and sampling setup
fs = 10000;                  % Sampling frequency
t = 0:1/fs:0.1;              % Time vector (100 ms)
N = length(t);               % Number of samples
f = linspace(-fs/2, fs/2, N); % Frequency vector for FFT

% Message signals
Am1 = 1;
fm1 = 50;                    % Message signal 1 frequency
msg1 = Am1 * cos(2*pi*fm1*t);

Am2 = 1;
fm2 = 120;                   % Message signal 2 frequency
msg2 = Am2 * cos(2*pi*fm2*t);

msg_total = msg1 + msg2;

% Carrier signal (LOWER frequency for visible envelope)
Ac = 1;                      % Carrier amplitude
fc = 1000;                    % Carrier frequency
carrier = Ac * cos(2*pi*fc*t);

% Modulation index < 1
mu = 1; %test all 3 cases

% AM Modulated Signals
AM_single = Ac * (1 + mu * msg1) .* cos(2*pi*fc*t);
AM_dual   = Ac * (1 + mu * msg_total) .* cos(2*pi*fc*t);

% FFT of AM signals
FFT_single = fftshift(abs(fft(AM_single))/N);
FFT_dual   = fftshift(abs(fft(AM_dual))/N);

clc;
clear;

% Time and frequency setup
fs = 10000;                 % Sampling frequency
t = 0:1/fs:0.1;             % Time vector (100 ms)
N = length(t);              % Number of samples
f = linspace(-fs/2, fs/2, N); % Frequency axis for FFT

% Message signal (single tone)
Am = 1;
fm = 50;
msg = Am * cos(2*pi*fm*t);  % Message signal

% Carrier signal
Ac = 1;
fc = 200;
carrier = Ac * cos(2*pi*fc*t);

% Modulation index beta = 1
beta = 1;
kf = beta * fm / Am;        % Frequency deviation constant

% Frequency Modulated signal
int_msg = cumsum(msg)/fs;   % Numerical integration (∫m(t) dt)
FM = Ac * cos(2*pi*fc*t + 2*pi*kf*int_msg);  % FM signal

% FFT of FM signal
FM_fft = fftshift(abs(fft(FM))/N);

% Plotting
figure;

subplot(2,2,1);
plot(t, msg);
title('Message Signal (Single Tone)');
xlabel('Time (s)');
ylabel('Amplitude');
grid on;

plot(t, carrier);
title('Carrier Signal');

ylabel('Amplitude');
